**Middleware helper — verify GESP_API_KEY for internal endpoints:**
```typescript
const GESP_API_KEY = process.env.GESP_API_KEY || "dev-key";

async function verifyApiKey(c: Context, next: () => Promise<void>) {
  const authHeader = c.req.header("Authorization");
  if (!authHeader || !authHeader.startsWith("Bearer ")) {
    return unauthorized(c, "Missing API key");
  }
  const token = authHeader.slice(7);
  if (token !== GESP_API_KEY) {
    return unauthorized(c, "Invalid API key");
  }
  await next();
}
```

```typescript
const startSchema = z.object({
  course_id: z.string().default("cpp"),
  start_level: z.number().int().min(1).max(8),

app.post("/start",
  StudentAuth(),
  zValidator("json", startSchema),
  async (c) => {
    const user = c.get("user");
    const body = c.req.valid("json");
    // 1. Create assessment session in DB
    const session = await assessment.createAssessmentSession({
      student_id: user.id,
      course_id: body.course_id,
      start_level: body.start_level,
      config_question_limit: body.config_question_limit,
      config_time_limit_min: body.config_time_limit_min,
    });
    // 2. Generate JWT token (per D-06)
    const token = await assessment.generateToken({
      assessment_session_id: session.id,
      student_id: user.id,
      course_id: body.course_id,
    });
    // Update session with token
    await db.update(assessmentSessions).set({ token }).where(eq(assessmentSessions.id, session.id));
    // 3. Create ellamaka session with agent="assessor" (per D-02)
    let ellamakaSessionId: string | null = null;
    try {
      const es = await ellamakaClient.createSession(
        `Assessment-${session.id.slice(0, 8)}`,
        "assessor"
      );
      ellamakaSessionId = es.id;
      await db.update(assessmentSessions)
        .set({ ellamaka_session_id: ellamakaSessionId })
        .where(eq(assessmentSessions.id, session.id));
      // 4. Send async prompt with system context (per D-02)
      const systemPrompt = buildSystemPrompt(token, user, body);
      await ellamakaClient.promptAsync(ellamakaSessionId, [
        { type: "text", text: `开始测评。学员 ${user.display_name}，课程 ${body.course_id}，起始级别 ${body.start_level}。请调用 get_question_candidates 获取候选题目并选择第一道题。` }
      ], systemPrompt);
    } catch (err) {
      logger.warn({ err, session_id: session.id }, "Ellamaka session creation failed, assessment in offline mode");

    }
    // 5. Wait for first question (15s timeout, per D-13 10s agent + 5s buffer)
    const firstQuestion = await waitForFirstQuestion(session.id, 15000);
    if (firstQuestion) {
```

```typescript
app.get("/next-question", async (c) => {
  const token = c.req.query("token");
  if (!token) return error(c, "Token required", 400);
```

**Endpoint 6: GET /{token}/stream — SSE stream (D-01):**
```typescript
app.get("/:token/stream", async (c) => {
  const token = c.req.param("token");
  let payload: TokenPayload;
  try { payload = await assessment.verifyToken(token); }
  catch { return unauthorized(c, "Invalid or expired token"); }
  const session = await db.query.assessmentSessions.findFirst({
    where: eq(assessmentSessions.id, payload.assessment_session_id),
  });
  if (!session?.ellamaka_session_id) {
    return error(c, "No active agent session", 404);
  }
  const stream = new ReadableStream({
    async start(controller) {
      try {
        for await (const event of ellamakaClient.streamEvents(session.ellamaka_session_id!)) {
          // Filter: only forward agent text output (not internal tool calls)
          if (
            (event.type === "message.part.delta" && event.field === "text") ||
            (event.type === "message.part.updated" && event.field === "text")
          ) {
            const text = event.delta || event.text || "";
            controller.enqueue(
              new TextEncoder().encode(`data: ${JSON.stringify({ type: "agent_text", text })}\n\n`)
            );
          }
        }
      } catch (err) {
        logger.error({ err, session_id: session.id }, "SSE stream error");
      }
      controller.close();
    },
  });
  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      "Connection": "keep-alive",
    },
  });
});
```

**Helper: question locks map + auto-select (D-13 10s fallback):**
```typescript
const questionLocks = new Map<string, string>(); // sessionId → questionId
const autoSelectTimers = new Map<string, ReturnType<typeof setTimeout>>();

// D-13: If agent doesn't call /select within 10s of receiving candidates,
// backend auto-selects the first candidate as fallback.
function startAutoSelectTimer(sessionId: string, fallbackQuestionId: string) {
  clearAutoSelectTimer(sessionId);
  const timer = setTimeout(async () => {
    logger.warn({ session_id: sessionId, question_id: fallbackQuestionId },
      "Agent did not select within 10s, auto-selecting first candidate (D-13 fallback)");
    await assessment.lockQuestion(sessionId, fallbackQuestionId);
    autoSelectTimers.delete(sessionId);
  }, 10_000); // D-13: 10s fallback
  autoSelectTimers.set(sessionId, timer);
}

function clearAutoSelectTimer(sessionId: string) {
  const existing = autoSelectTimers.get(sessionId);
  if (existing) {
    clearTimeout(existing);
    autoSelectTimers.delete(sessionId);
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → POST /start, /submit, /resume | Student-facing endpoints require session auth + JWT token |
| Client → GET /next-question, /progress, /stream | Token-based endpoints verify JWT before returning data |
| gesp-plugin → POST /candidates, /select, /evaluate | Internal endpoints require GESP_API_KEY bearer auth |
| Backend → ellamaka | Calls to ellamaka via EllamakaClient (internal service boundary) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-11 | Spoofing | All token-based endpoints | mitigate | Token verification via hono/jwt `verify()` on every request. Short 2h expiry limits replay window. |
| T-03-12 | Spoofing | Internal endpoints (/candidates, /select, /evaluate) | mitigate | GESP_API_KEY bearer token verified in verifyApiKey middleware. Constant-time comparison (strict equality). |
| T-03-13 | Information Disclosure | GET /next-question | mitigate | Returns full question content only after agent has locked the question via /select. Before lock, returns { waiting: true }. Question content is only available via token-authenticated endpoint. |
| T-03-14 | Information Disclosure | POST /candidates | mitigate | Returns candidates with `short_summary` only (100 chars of content). Never returns full question content or answer. Anti-leak design per D-13. |
| T-03-15 | Tampering | POST /submit | mitigate | Token verified before processing answer. student_id in token must match authenticated user (additional check: payload.student_id === user.id). Answer is stored with session_id from token — can't submit answer to another student's session. |
| T-03-16 | Denial of Service | GET /{token}/stream | accept | SSE connections are per-session with token verification. No rate limiting on connections — acceptable for MVP single-user scenario. |
| T-03-17 | Information Disclosure | Agent internal messages | mitigate | Agent notification messages (answer results, coding submissions) are sent via ellamaka promptAsync — these are internal messages not visible to the student's chat panel. Frontend SSE only forwards events where field="text" — tool call events are filtered out. |
</threat_model>
