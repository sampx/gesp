3. Build embedding text per table type (private method):
```typescript
private buildEmbeddingText(table: KnowledgeTable, data: Record<string, unknown>): string {
  switch (table) {
    case "points": {
      const point = (data.point as string) || "";
      const desc = (data.description as string) || "";
      return desc ? `${point}: ${desc}` : point;
    }
    case "lessons": {
      const title = (data.title as string) || "";
      const objectives = Array.isArray(data.objectives) ? data.objectives.join(" ") : "";
      const keyPoints = Array.isArray(data.key_points) ? data.key_points.join(" ") : "";
      return `${title} ${objectives} ${keyPoints}`;
    }
    case "questions": {
      const content = (data.content as string) || "";
      const explanation = (data.explanation as string) || "";
      const hints = Array.isArray(data.hints) ? data.hints.join(" ") : "";
      const mistakes = Array.isArray(data.common_mistakes) ? data.common_mistakes.join(" ") : "";
      const parts = [content, explanation, hints, mistakes].filter(Boolean);
      return parts.length > 0 ? parts.join(" ") : content;
    }
    case "exams": {
      const content = (data.content as string) || "";
      const explanation = (data.explanation as string) || "";
      return explanation ? `${content}: ${explanation}` : content;
    }
  }
}
```

```typescript
// Create/Update knowledge point
const pointSchema = z.object({
  level: z.number().int().min(1).max(8),
  block: z.string().min(1),
  point: z.string().min(1),
  mastery_verb: z.enum(["了解", "理解", "掌握"]).default("了解"),
  description: z.string().default(""),
  tags: z.array(z.string()).default([]),
  language: z.string().default("C++"),
  block_order: z.number().int().default(1),
  point_order: z.number().int().default(1),
});

const searchSchema = z.object({
  query: z.string().min(1),
  limit: z.coerce.number().default(10),
  filter: z.string().optional(),
});

const paginationSchema = z.object({
  page: z.coerce.number().int().min(1).default(1),
  limit: z.coerce.number().int().min(1).max(100).default(20),
});
```

3. Student route (public authenticated):
```typescript
export const studentKnowledgeRouter = new Hono<Env>();
studentKnowledgeRouter.get(
  "/search",
  requireAuth,
  zValidator("query", searchSchema.pick({ query: true, limit: true })),
  async (c) => {
    const kb = c.get("knowledgeBaseService"); // assume injected via middleware
    const { query, limit = 5 } = c.req.valid("query");
    const results = await kb.search("points", query, Math.min(limit, 5));
    return c.json({ success: true, data: results });
  }
);
```

4. Admin routes (role-restricted):
```typescript
export const adminKnowledgeRouter = new Hono<Env>();
adminKnowledgeRouter.get(
  "/points",
  requireAuth,
  requireRole(["admin", "root"]),
  zValidator("query", paginationSchema),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    const { page, limit } = c.req.valid("query");
    const result = await kb.list("points", page, limit);
    return c.json({ success: true, data: result });
  }
);

adminKnowledgeRouter.get(
  "/points/search",
  requireAuth,
  requireRole(["admin", "root"]),
  zValidator("query", searchSchema),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    const { query, limit, filter } = c.req.valid("query");
    const results = await kb.search("points", query, limit, filter);
    return c.json({ success: true, data: results });
  }
);

adminKnowledgeRouter.get(
  "/points/:id",
  requireAuth,
  requireRole(["admin", "root"]),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    const result = await kb.get("points", c.req.param("id"));
    if (!result) return c.json({ success: false, message: "Not found" }, 404);
    return c.json({ success: true, data: result });
  }
);

adminKnowledgeRouter.post(
  "/points",
  requireAuth,
  requireRole(["admin", "root"]),
  zValidator("json", pointSchema),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    const data = c.req.valid("json");
    const result = await kb.create("points", data);
    return c.json({ success: true, data: result }, 201);
  }
);

adminKnowledgeRouter.put(
  "/points/:id",
  requireAuth,
  requireRole(["admin", "root"]),
  zValidator("json", pointSchema.partial()),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    const data = c.req.valid("json");
    const result = await kb.update("points", c.req.param("id"), data);
    return c.json({ success: true, data: result });
  }
);

adminKnowledgeRouter.delete(
  "/points/:id",
  requireAuth,
  requireRole(["admin", "root"]),
  async (c) => {
    const kb = c.get("knowledgeBaseService");
    await kb.delete("points", c.req.param("id"));
    return c.json({ success: true, message: "已删除" });
  }
);
```
